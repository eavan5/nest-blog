import { Meta } from './meta.schema';

@Schema()
export class Article {
  @Prop({
    type: String,
    trim: true,
  })
  title: string;

  @Prop()
  content: string;

  // @Prop()
  // author_id: string;

  @Prop()
  desc: string;

  @Prop()
  headPic: string;

  @Prop()
  add_time: number;

  @Prop()
  views: number;

  @Prop()
  update_time: number;

  // @Prop()
  // hidden: number;

  @Prop([])
  comment: [];


  @Prop([

  category_id: Meta[];

  @Prop([
    {
      type: mongoose.Schema.Types.ObjectId,
      ref: 'Meta',
